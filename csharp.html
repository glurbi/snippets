                    public IEnumerable<int> Fibo(int n)
                    {
                        yield return 1;
                        yield return 1;

                        var f2 = 1;
                        var f1 = 1;

                        for (int i = 0; i < n; i++)
                        {
                            var f0 = f1 + f2;
                            yield return f0;
                            f2 = f1;
                            f1 = f0;
                        }
                    }
