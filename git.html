GIT

Configuration Parameters

                
                    # list the global configuration parameters
                    git config --list --global

                    # list the repository specific configuration parameters
                    git config --list --local

                    # set the user email associated with the repository
                    git config user.email "foo@bar.com"

                    # set the user email globally
                    git config --global user.email "foo@bar.com"

Branch Management

                
                    # list all branches, local and remote
                    git branch --local

                    # create a branch and set it as the current working branch
                    git checkout -b <branch_name>

                    # which is a shorthand for
                    git branch <branch_name>
                    git checkout <branch_name>

                    # delete local branch if it has been fully merge with its upstream branch
                    git branch --delete <branch_name>
                    git branch -d <branch_name>

                    # delete local branch irrespectively of its merged status
                    git branch --delete --force <branch_name>
                    git branch -D <branch_name>

                    # delete remote branch
                    git push <remote_name> --delete <branch_name>
                    git push <remote_name> -d <branch_name>

                    # before fetching, remove any remote-tracking references that no longer exist on the remote
                    git fetch --prune
                    git fetch --all --prune

                    # gives details about local branches, in particular the upstream branch
                    git branch -vv
